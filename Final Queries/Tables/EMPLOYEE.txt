CREATE TABLE  "EMPLOYEE" 
   (	"E_ID" NUMBER(5,0), 
	"E_NAME" VARCHAR2(30) NOT NULL ENABLE, 
	"E_DOB" DATE NOT NULL ENABLE, 
	"E_GENDER" CHAR(1), 
	"E_ADD" VARCHAR2(50) NOT NULL ENABLE, 
	"E_EMAIL" VARCHAR2(50), 
	"E_DESIGNATION" VARCHAR2(20) NOT NULL ENABLE, 
	"E_PHONE" NUMBER(10,0), 
	 PRIMARY KEY ("E_ID") ENABLE
   )
