CREATE TABLE  "SUPPLIES" 
   (	"C_ID" NUMBER(5,0), 
	"S_DAY" NUMBER(2,0) NOT NULL ENABLE, 
	"S_MONTH" VARCHAR2(12) NOT NULL ENABLE, 
	"S_YEAR" NUMBER(4,0) NOT NULL ENABLE, 
	"A_ID" NUMBER(5,0), 
	 FOREIGN KEY ("A_ID")
	  REFERENCES  "AGENT" ("A_ID") ENABLE, 
	 FOREIGN KEY ("C_ID")
	  REFERENCES  "RAWCOTTON" ("C_ID") ENABLE
   )
